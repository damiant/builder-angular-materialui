# BottomSheet Components

## Overview

### Purpose

Use this component to generate bottom sheet functionality. The BottomSheet component provides Material Design styled modal panels that slide up from the bottom of the screen to display contextual content, actions, or detailed information without navigating away from the current page. It follows Material Design 3 specifications for bottom sheets, offering consistent animations, accessibility features, and flexible content presentation for mobile-first and responsive design patterns.

### When to Use

- **Contextual Actions**: When you need to provide actions related to the current screen content without full page navigation
- **Mobile Navigation**: For mobile-optimized navigation menus and option lists that slide up from the bottom
- **Content Previews**: When displaying detailed information or previews that complement the main content
- **Form Input**: For collecting user input in a focused overlay without losing current page context
- **Settings and Options**: When providing quick access to settings, filters, or configuration options
- **Share and Export**: For sharing content or export options that need immediate access
- **Image and Media Viewers**: When displaying media content with associated controls and information
- **Quick Actions**: For providing quick actions on list items or cards without full dialog overhead
- **Confirmation Flows**: When requiring user confirmation for actions in a less intrusive manner than dialogs
- **Progressive Disclosure**: For revealing additional information or options progressively as users interact with content

### Architecture

**No Coupling**: The BottomSheet component is a standalone service-based component that operates independently without requiring other components. It provides:

- **Service-based instantiation**: Complete bottom sheet implementation accessible through injectable service
- **Overlay management**: Automatic positioning, backdrop handling, and lifecycle management
- **Flexible content**: Support for both component-based and template-based content
- **Global accessibility**: Consistent behavior and styling across the entire application
- **Independent operation**: No dependencies on parent or child components for basic functionality

## Components

### MatBottomSheet {Service Component}

A versatile service component that creates and manages bottom sheet overlays with Material Design animations, backdrop handling, and comprehensive configuration options for various content presentation scenarios.

**Purpose:** Injectable service that creates modal bottom sheet overlays with Material Design styling, proper accessibility features, responsive behavior, and support for both simple content and complex interactive components.

**Interface:**

```typescript
interface MatBottomSheet {
  // Content presentation methods
  open<T, D = any, R = any>(
    component: ComponentType<T>,
    config?: MatBottomSheetConfig<D>,
  ): MatBottomSheetRef<T, R>;
  open<T, D = any, R = any>(
    template: TemplateRef<T>,
    config?: MatBottomSheetConfig<D>,
  ): MatBottomSheetRef<T, R>;

  // Management methods
  dismiss<R = any>(result?: R): void;
  ngOnDestroy(): void;
}

interface MatBottomSheetConfig<D = any> {
  // Content and Data
  data?: D | null; // Data to inject into the bottom sheet content
  viewContainerRef?: ViewContainerRef; // Container for dependency injection context

  // Styling and Layout
  panelClass?: string | string[]; // Custom CSS classes for styling
  backdropClass?: string; // Custom class for the backdrop
  height?: string; // Fixed height for the bottom sheet
  minHeight?: number | string; // Minimum height (number assumes pixels)
  maxHeight?: number | string; // Maximum height (number assumes pixels)

  // Behavior Configuration
  hasBackdrop?: boolean; // Whether to show backdrop (default: true)
  disableClose?: boolean; // Prevent closing via backdrop click or escape
  closeOnNavigation?: boolean; // Auto-close on navigation changes
  restoreFocus?: boolean; // Restore focus to trigger element on close
  scrollStrategy?: ScrollStrategy; // How to handle scrolling while open

  // Accessibility
  ariaLabel?: string | null; // Accessible label for screen readers
  ariaModal?: boolean; // Whether to set aria-modal attribute
  autoFocus?: AutoFocusTarget | string | boolean; // Focus management on open

  // Layout Direction
  direction?: Direction; // Text direction for RTL support
}

interface MatBottomSheetRef<T = any, R = any> {
  // Component access
  instance: T; // Reference to the content component instance
  componentRef: ComponentRef<T> | null; // Component reference (null for templates)
  containerInstance: MatBottomSheetContainer; // Container component reference

  // State management
  disableClose: boolean | undefined; // Whether closing is disabled

  // Lifecycle methods
  dismiss(result?: R): void; // Programmatically close the bottom sheet

  // Event observables
  afterDismissed(): Observable<R | undefined>; // Fires when closed with optional result
  afterOpened(): Observable<void>; // Fires when open and animations complete
  backdropClick(): Observable<MouseEvent>; // Fires when backdrop is clicked
  keydownEvents(): Observable<KeyboardEvent>; // Keyboard events targeted at overlay
}

type AutoFocusTarget = "dialog" | "first-tabbable" | "first-heading";
```

```typescript
  contactForm: FormGroup;
  isSubmitting = false;

  constructor(
    private bottomSheetRef: MatBottomSheetRef<FormBottomSheetComponent>,
    @Inject(MAT_BOTTOM_SHEET_DATA) public data: any,
    private fb: FormBuilder,
  ) {
    this.contactForm = this.fb.group({
      name: [data.contact?.name || "", [Validators.required]],
      email: [
        data.contact?.email || "",
        [Validators.required, Validators.email],
      ],
      phone: [data.contact?.phone || ""],
      subject: [data.contact?.subject || "general", [Validators.required]],
      message: [
        data.contact?.message || "",
        [Validators.required, Validators.minLength(10)],
      ],
    });
  }

  async onSubmit(): Promise<void> {
    if (this.contactForm.valid && !this.isSubmitting) {
      this.isSubmitting = true;

      try {
        // Simulate API call
        await new Promise((resolve) => setTimeout(resolve, 2000));

        this.bottomSheetRef.dismiss({
          action: "submit",
          formData: this.contactForm.value,
        });
      } catch (error) {
        console.error("Form submission error:", error);
        this.isSubmitting = false;
      }
    }
  }

  dismiss(): void {
    if (this.contactForm.dirty && !this.isSubmitting) {
      // Could show confirmation dialog here
      if (
        confirm("You have unsaved changes. Are you sure you want to close?")
      ) {
        this.bottomSheetRef.dismiss();
      }
    } else {
      this.bottomSheetRef.dismiss();
    }
  }
}
```

## Best Practices

### Usage Guidelines

- **Design for mobile-first**: Bottom sheets are primarily mobile interaction patterns, ensure they work well on small screens
- **Keep content focused**: Limit bottom sheet content to essential information and actions to maintain clarity
- **Provide clear escape paths**: Always include obvious ways to close the bottom sheet (close button, backdrop click, or swipe gesture)
- **Use appropriate heights**: Set reasonable min/max heights to ensure content is accessible without overwhelming the screen
- **Handle keyboard interactions**: Ensure proper focus management and keyboard navigation within bottom sheet content
- **Consider content hierarchy**: Use clear visual hierarchy with headers, sections, and actions to guide user attention

### Accessibility

- **Focus management**: Ensure focus moves to the bottom sheet when opened and returns appropriately when closed
- **Keyboard navigation**: Support Tab navigation within the bottom sheet and Escape key for closing
- **Screen reader support**: Use proper ARIA labels and roles to announce bottom sheet content and purpose
- **High contrast support**: Ensure sufficient color contrast for all text and interactive elements
- **Touch targets**: Maintain adequate touch target sizes (minimum 44px) for mobile interactions
- **Backdrop behavior**: Clearly communicate backdrop click behavior and provide alternative close methods

### Performance

- **Lazy load content**: Consider lazy loading complex bottom sheet components to improve initial app performance
- **Optimize animations**: Use CSS transforms and GPU-accelerated properties for smooth slide animations
- **Memory management**: Properly clean up subscriptions and event listeners when bottom sheet is dismissed
- **Content virtualization**: For long lists in bottom sheets, implement virtual scrolling to maintain performance
- **Debounce interactions**: Prevent rapid successive bottom sheet openings that could impact performance
- **Minimize DOM manipulation**: Keep bottom sheet content lightweight and avoid expensive DOM operations

### Pattern 2: Responsive Bottom Sheet with Breakpoint Handling

```typescript
import { Injectable, inject } from "@angular/core";
import { BreakpointObserver, Breakpoints } from "@angular/cdk/layout";
import { MatBottomSheet } from "@angular/material/bottom-sheet";
import { MatDialog } from "@angular/material/dialog";
import { Observable } from "rxjs";
import { map } from "rxjs/operators";

@Injectable({
  providedIn: "root",
})
export class ResponsiveSheetService {
  private bottomSheet = inject(MatBottomSheet);
  private dialog = inject(MatDialog);
  private breakpointObserver = inject(BreakpointObserver);

  private isMobile$ = this.breakpointObserver
    .observe([Breakpoints.Handset, Breakpoints.Small])
    .pipe(map((result) => result.matches));

  openResponsive<T, R = any>(component: any, config?: any): Observable<R> {
    return new Observable((observer) => {
      this.isMobile$
        .subscribe((isMobile) => {
          if (isMobile) {
            // Use bottom sheet for mobile
            const bottomSheetRef = this.bottomSheet.open(component, {
              ...config,
              panelClass: [...(config?.panelClass || []), "mobile-sheet"],
            });

            bottomSheetRef.afterDismissed().subscribe((result) => {
              observer.next(result);
              observer.complete();
            });
          } else {
            // Use dialog for desktop
            const dialogRef = this.dialog.open(component, {
              ...config,
              width: config?.width || "500px",
              maxWidth: "90vw",
              panelClass: [...(config?.panelClass || []), "desktop-dialog"],
            });

            dialogRef.afterClosed().subscribe((result) => {
              observer.next(result);
              observer.complete();
            });
          }
        })
        .unsubscribe();
    });
  }

  // Specialized responsive methods
  openResponsiveActionSheet(actions: any[]): Observable<string> {
    return this.openResponsive(ActionSheetComponent, {
      data: { actions },
      panelClass: ["action-sheet"],
    });
  }

  openResponsiveForm(formComponent: any, data?: any): Observable<any> {
    return this.openResponsive(formComponent, {
      data,
      disableClose: true,
      panelClass: ["form-container"],
    });
  }

  openResponsiveImageViewer(images: string[], index = 0): Observable<void> {
    return this.isMobile$.pipe(
      map((isMobile) => {
        if (isMobile) {
          return this.bottomSheet.open(ImageViewerSheetComponent, {
            data: { images, index },
            panelClass: ["image-viewer-sheet"],
            height: "100vh",
          });
        } else {
          return this.dialog.open(ImageViewerDialogComponent, {
            data: { images, index },
            panelClass: ["image-viewer-dialog"],
            width: "80vw",
            height: "80vh",
            maxWidth: "1200px",
            maxHeight: "800px",
          });
        }
      }),
    );
  }
}
```
